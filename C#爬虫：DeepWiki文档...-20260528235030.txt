五、关键技术点详解
URL 转换与验证
public class UrlConverter
{
    public static string ToDeepWiki(string githubUrl)
    {
        // 支持多种格式: 
        // https://github.com/owner/repo
        // https://github.com/owner/repo.git
        // github.com/owner/repo/tree/main/src
        var match = Regex.Match(githubUrl, @"github\.com/([^/]+)/([^/.]+)");
        if (!match.Success) throw new ArgumentException("无效的GitHub URL");
        return $"https://deepwiki.com/{match.Groups[1].Value}/{match.Groups[2].Value}";
    }
}
侧边栏解析（最关键的一步）
⚠️ 需要你实际打开一个 DeepWiki 页面，F12 确认侧边栏的 CSS Selector。根据搜索结果，DeepWiki 页面 URL 形如 /owner/repo/2-core-architecture，侧边栏包含带编号的层级链接。

public class SidebarParser
{
    // 需要根据实际DOM调整选择器
    private const string SidebarSelector = "nav a[href], aside a[href]"; 
    
    public async Task<List<WikiPage>> ParseAsync(IDocument document, string baseUrl)
    {
        var links = document.QuerySelectorAll(SidebarSelector);
        var pages = new List<WikiPage>();
        
        foreach (var link in links)
        {
            var href = link.GetAttribute("href");
            if (string.IsNullOrEmpty(href) || !href.StartsWith("/")) continue;
            
            pages.Add(new WikiPage
            {
                Title = link.TextContent.Trim(),
                Slug = href.Split('/').Last(),
                FullUrl = $"https://deepwiki.com{href}",
                Depth = CountDepth(href) // 根据URL路径深度判断层级
            });
        }
        return pages.DistinctBy(p => p.FullUrl).ToList();
    }
}
双模式内容获取（应对 SPA）
public class PageFetcher
{
    private readonly HttpClient _http;
    private readonly IPlaywright? _playwright; // nullable, 按需启用
    
    public async Task<string> FetchAsync(string url, bool forceBrowser = false)
    {
        if (!forceBrowser)
        {
            var html = await _http.GetStringAsync(url);
            // 检查是否包含实际内容（SSR 场景）
            if (ContainsMainContent(html)) return html;
        }
        
        // Fallback: Playwright 渲染
        using var browser = await _playwright!.Chromium.LaunchAsync(new() { Headless = true });
        using var page = await browser.NewPageAsync();
        await page.GotoAsync(url, new() { WaitUntil = WaitUntilState.NetworkIdle });
        await page.WaitForSelectorAsync("article, main, [class*='content']");
        return await page.ContentAsync();
    }
}

八、⚠️ 重要注意事项
先做 P0 验证：DeepWiki 是 SPA，必须先确认纯 HTTP 请求能否获得完整 HTML。如果内容是纯客户端渲染的，则必须上 Playwright，架构复杂度会显著增加。
robots.txt：爬取前先检查 https://deepwiki.com/robots.txt，遵守其规则。
频率控制：建议默认 2-3 秒/请求，避免对服务造成压力。DeepWiki 是免费服务，过度爬取可能导致 IP 被封。
缓存机制：对已下载的页面做本地缓存（SQLite/文件哈希），支持中断后继续爬取，避免重复请求。
法律合规：仅用于个人学习研究，不要重新分发或商业化使用爬取的内容。
关注 deepwiki-open：搜索结果显示存在开源版 deepwiki-open，它可能有更直接的数据获取方式，值得研究其源码作为参考。
